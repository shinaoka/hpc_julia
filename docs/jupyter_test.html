function compute_m_120degrees2(spins)
    cos_sum = sum([cos(3*atan(x[2], x[1])) for x in spins])
    sin_sum = sum([sin(3*atan(x[2], x[1])) for x in spins])
    (cos_sum^2 + sin_sum^2)/length(spins)^2
end
